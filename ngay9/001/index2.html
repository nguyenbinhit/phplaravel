tham số của hàm
        đối khi gọi hàm
        tham số của function

        trong ví dụ tinhTong  thì a và b là tham số (parameter)
        function tinhTong (a, b) {
            // 2 tham số a và b chúng ta có thể sử dụng nó như biến trong function
        }

        -đối số
        khi gọi hàm var c = 2, d = 3
        tinhTong(c, d);
        lúc này c và d là 2 đối số được truyền cho function
